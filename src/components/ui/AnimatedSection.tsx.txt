import React from 'react';

interface AnimatedSectionProps extends React.HTMLAttributes<HTMLDivElement> {
  // O componente Hero só passa 'className'
  children: React.ReactNode;
  className?: string;
}

// Versão básica: Apenas um wrapper <div> que não aplica animação
// (para evitar dependências de bibliotecas como framer-motion)
const AnimatedSection: React.FC<AnimatedSectionProps> = ({ children, className, ...props }) => {
  return (
    <div className={className} {...props}>
      {children}
    </div>
  );
};

export default AnimatedSection;
